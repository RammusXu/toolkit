    rs.initiate(
  {
    _id: "<replSetName>",
    configsvr: true,
    members: [
      { _id : 0, host : "cfg1.example.net:27019" },
      { _id : 1, host : "cfg2.example.net:27019", priority : 0.5 },
      { _id : 2, host : "cfg3.example.net:27019", priority : 0.5 }
    ]
  }
)
